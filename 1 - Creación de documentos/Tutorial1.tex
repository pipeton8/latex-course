  % Primero, debemos declarar la clase de documento que queremos escribir:
\documentclass{article}

% Hay distintas clases de documento, las cuales tienen diferentes opciones predeterminadas. 
% Una de ellas (y la más usada) es ``article'', pero también está ``book'', ``report'', ``letter'', etc.
% La clase ``beamer'', es aquella con la que se hacen presentaciones.
% Es posible crear una clase propia (por ejemplo, yo tengo mis clases ``apunte'', ``tarea'' y ``ayudantía'',
% cada una con sus propiedades particulares).

% Luego de declarar el tipo de documento debemos ``equipar'' nuestro texto con las herramientas
% que vamos a utilizar. Esto incluye tanto los elementos para incorporar fórmulas matemáticas como
% aquellos para manejar los acentos, el idioma, las figuras e incluso el tipo de letra. 
% Estas herramientas se llaman PAQUETES y se incluyen de la siguiente manera:
%	\usepackage[opciones]{nombre del paquete}

% La mayoría de los paquetes importantes requieren opciones, pero como veremos más adelante no siempre es así.

% Agregamos ahora los paquetes:
\usepackage[utf8]{inputenc} % Este paquete en particular maneja la codificación del texto. Para efectos prácticos permite usar acentos en el texto.

\usepackage[spanish]{babel} % Este paquete se encarga del idioma, así las demostraciones dicen ``Demostración'' y no ``Proof''.

\usepackage{mathtools, amsfonts, amsmath, amsthm, amssymb} % Múltiples paquetes para cosas matemáticas.

% Por ahora usaremos solo estos paquetes.

% Hasta ahora no hemos escrito nada del documento, todo lo que hemos escrito hasta ahora se llama ``Preámbulo'' 
% y es una de las estructuras centrales de un documento de LaTeX. El preámbulo también incorpora otros elementos,
% como los márgenes de la página o 
% Evidentemente, a medida que los textos se complejizan, la cantidad de paquetes que se necesitan crece bastante.
% Es por esto que a veces es conveniente crear paquetes propios, que resuman lo más posible el preámbulo (por ejemplo,
% yo tengo mi paquete ``felipito'').

% Ahora queremos escribir algo. Partiremos con algo similar a ``Hola mundo''
% Para comenzar a escribir el texto, después del preámbulo le avisamos a LaTeX que empezaremos, escribiendo:
\begin{document}

% Ahora escribimos el texto
!`Hola a todos! ?`cómo están? Así de fácil es escribir en \LaTeX

% Finalmente, después de la última linea que escribamos, avisamos que terminamos escribiendo
\end{document}

% Ahora estamos listos para ``compilar'', es decir, para transformar este código en un PDF.
